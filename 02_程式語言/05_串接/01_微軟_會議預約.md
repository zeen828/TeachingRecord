# 微軟行事曆

這是透過微軟 Graph API去預約行事曆在通知信賤開啟Teams Meeting

```php
<?php

namespace App\Libraries\microsoft;

use GuzzleHttp\Client;
// Lib
use Exception;
use Illuminate\Support\Facades\Log;
use Illuminate\Support\Facades\Cache;
use anlutro\LaravelSettings\Facade as Setting;
use App\Libraries\LibBase;

/**
 * 微軟-會議室通知信
 */
class GraphTeams extends LibBase
{
    protected $client;
    protected $base_uri;
    protected $access_token;
    protected $refresh_token;

    public function __construct()
    {
        parent::__construct();

        $this->type_id = 4;
        // $this->base_uri = env('AZURE_GRAPH_BASE_URI', 'https://login.microsoftonline.com');

        $this->client = new Client([
            // 'base_uri' => $this->base_uri,
            // 'timeout'  => 30,
            'verify'   => false, // <- 忽略憑證
            // 'http_errors' => false,
        ]);
    }

    public function saveTokens($accessToken, $expiresIn)
    {
        $ttl = max(1, floor(($expiresIn - 60) / 60)); // 最少 1 分鐘，減掉 60 秒 buffer
        Cache::put('azure_access_token', $accessToken, $ttl);
    }

    public function getAccessToken()
    {
        return Cache::get('azure_access_token');
    }

    public function clearTokens()
    {
        Cache::forget('azure_access_token');
    }

    /**
     * 檢查token
     */
    public function checkAuth()
    {
        $access_token = $this->getAccessToken();
        if (empty($access_token)) {
            $json = $this->loginApplicationToken();
            return isset($json['access_token']);
        } else {
            // access_token時效內
            $this->access_token = $access_token;
            return true;
        }
    }

    /**
     * 取得串接Token
     */
    public function loginApplicationToken()
    {
        $title = '微軟 Graph API 取Token';
        Log::info(sprintf('[Lib - %s] %s, function %s()', $title, __CLASS__, __FUNCTION__));

        $uri = sprintf('%s/%s/oauth2/v2.0/token', env('AZURE_GRAPH_BASE_URI', 'https://login.microsoftonline.com'), env('AZURE_TENANT_ID'));

        $formData = [
            'form_params' => [
                'client_id'     => env('AZURE_CLIENT_ID'),
                'client_secret' => env('AZURE_CLIENT_SECRET'),
                'scope'         => 'https://graph.microsoft.com/.default',
                'grant_type'    => 'client_credentials',
            ],
            'http_errors' => false,   // 先不要拋 exception
        ];

        try {
            $response = $this->client->post($uri, $formData);

            $code = $response->getStatusCode();
            $body = $response->getBody();
            $jsonData = json_decode($body, true);

            if ($code >= 200 && $code < 300) {
                $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $body, $code, true);

                // 做心跳
                Setting::set('microsoft.teams.lib', date('H:i:s'));
                Setting::save();

                if (isset($jsonData['access_token']) && isset($jsonData['expires_in'])) {
                    $this->saveTokens($jsonData['access_token'], $jsonData['expires_in']);
                }
                return $jsonData;
            } else {
                $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $body, $code, false);

                return $jsonData;
            }
        } catch (Exception $e) {
            $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $e->getMessage(), '999', false);

            return ['error' => $e->getMessage()];
        }
    }

    public function createEventTeamsMeeting($main_email, $subject, $star_date, $end_date, $attendees_array, $body_htnl='')
    {
        $title = '微軟 Graph API 建立Teams 會議室';
        Log::info(sprintf('[Lib - %s] %s, function %s()', $title, __CLASS__, __FUNCTION__));

        $auth = $this->checkAuth();
        if (empty($auth)) {
            return ['error' => 'Unable to get access token'];
        }

        // 檢查信箱是否光寶人員
        if (substr($main_email, -11) === '@liteon.com') {
            $uri = sprintf('https://graph.microsoft.com/v1.0/users/%s/calendar/events', $main_email);
        } else {
            $uri = sprintf('https://graph.microsoft.com/v1.0/users/%s/calendar/events', 'zhspace.test@liteon.com');    
        }

        $payload = [
            'subject' => $subject,
            'start' => [
                'dateTime' => $star_date,
                'timeZone' => 'Asia/Taipei',
            ],
            'end' => [
                'dateTime' => $end_date,
                'timeZone' => 'Asia/Taipei',
            ],
            'isOnlineMeeting' => true,
            'onlineMeetingProvider' => 'teamsForBusiness',
            'attendees' => $attendees_array,
            // 'attendees' => [
            //     [
            //         'emailAddress' => [
            //             'address' => 'will@nsst.com.tw',
            //             'name' => '客戶代表',
            //         ],
            //         'type' => 'required',
            //     ]
            // ],
            'body' => [
                'contentType' => 'HTML',
                'content' => $body_htnl,
            ],
        ];

        $formData = [
            'headers' => [
                'Authorization' => sprintf('Bearer %s', $this->getAccessToken()),
                'Content-Type'  => 'application/json',
            ],
            'json' => $payload,
            'http_errors' => false,   // 先不要拋 exception
        ];

        // print_r(json_encode($formData));
        // exit();
        try {
            $response = $this->client->post($uri, $formData);

            $code = $response->getStatusCode();
            $body = $response->getBody();
            $jsonData = json_decode($body, true);

            print_r($jsonData);

            if ($code >= 200 && $code < 300) {
                $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $body, $code, true);

                return $jsonData;
            } else {
                $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $body, $code, false);

                return $jsonData;
            }
        } catch (Exception $e) {
            $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $e->getMessage(), '999', false);

            return ['error' => $e->getMessage()];
        }
    }

    public function createChat($topic = null)
    {
        $title = '微軟 Graph API 建立Chat';
        Log::info(sprintf('[Lib - %s] %s, function %s()', $title, __CLASS__, __FUNCTION__));

        $auth = $this->checkAuth();
        if (empty($auth)) {
            return ['error' => 'Unable to get access token'];
        }

        // $uri = sprintf('/%s/oauth2/v2.0/token', env('AZURE_TENANT_ID'));
        $uri = 'https://graph.microsoft.com/v1.0/chats';

        $payload = [
            'chatType' => 'group',
            'topic'    => (empty($topic))? sprintf('[%s]會議室預約系統', date('Y-m-d')) : $topic,
            'members'  => [
                [
                    '@odata.type' => '#microsoft.graph.aadUserConversationMember',
                    'roles' => ['owner'],
                    'user@odata.bind' =>
                        "https://graph.microsoft.com/v1.0/users('zhspace.test@liteon.com')"
                ]
            ]
        ];

        $formData = [
            'headers' => [
                'Authorization' => sprintf('Bearer %s', $this->getAccessToken()),
                'Content-Type'  => 'application/json',
            ],
            'json' => $payload,
            'http_errors' => false,   // 先不要拋 exception
        ];

        try {
            $response = $this->client->post($uri, $formData);

            $code = $response->getStatusCode();
            $body = $response->getBody();
            $jsonData = json_decode($body, true);

            // print_r($jsonData);

            if ($code >= 200 && $code < 300) {
                $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $body, $code, true);

                return $jsonData;
            } else {
                $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $body, $code, false);

                return $jsonData;
            }
        } catch (Exception $e) {
            $this->WriteApiLog($title, 'POST', $this->base_uri, $uri, $formData, $e->getMessage(), '999', false);

            return ['error' => $e->getMessage()];
        }
    }
}
```
