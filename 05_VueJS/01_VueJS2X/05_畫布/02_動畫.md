# 動畫

## **敘述**
畫布可以繪製一些簡單的動畫，我們可以查詢到HTML下的教學但比較少使用在Vue.js的教學，這邊將以Vue.js為範例。

## **教學&範例**
## 基礎
```vue
<template>
  <canvas id="canvas" class="canvas" ref="canvas" :width="width" :height="height">畫布</canvas>
</template>

<script>
export default {
  name: 'canvas-demo',
  methods: {
    ready () {
      // 畫布宣告
      this.initCanvas()
      // 繪圖
      this.drawing()
    },
    // 畫布宣告
    initCanvas () {
      // 定位目標DOM
      this.canvas = document.getElementById('canvas')
      // 渲染
      if (this.canvas.getContext) {
        this.context = this.canvas.getContext('2d')
      } else {
        console.log('瀏覽器不支援畫布(canvas)')
        alert('瀏覽器不支援畫布(canvas)')
      }
    },
    // 繪圖
    drawing () {
      // 迴圈呼叫函式
      this.myLoop = setInterval(this.setRound, this.speed)
    },
    // 畫圓形
    setRound () {
      // 清空畫布
      this.context.clearRect(0, 0, this.canvas.width, this.canvas.height)
      // 開始一個新路徑
      this.context.beginPath()
      // 產生一個弧形(圓心座標X,圓心座標Y,半徑,開始點,結束點,true逆向)
      this.context.arc(50 + this.x, 200, 20, 0, Math.PI * 2, true)
      // 結束一個路徑
      this.context.closePath()
      // 填滿路徑內容
      this.context.fill()
      // 移動速度看這個
      this.x += 10
      if (this.x > this.width - 100) {
        // 停止迴圈
        clearInterval(this.myLoop)
      }
    }
  },
  data () {
    return {
      // 畫布
      canvas: this.$refs.canvas,
      context: null,
      // 寬高
      width: 400,
      height: 400,
      x: 0,
      y: 0,
      // 動畫畫面刷新速度
      speed: 30,
      // 迴圈
      myLoop: null
    }
  },
  mounted () {
    this.ready()
  },
  beforeDestroy () {
    // 停止迴圈
    clearInterval(this.myLoop)
  }
}
</script>
```
